#include <iostream>
#include <vector>
using namespace std;

int main() {
    int n;
    cout << "Enter number of courses taken: ";
    cin >> n;

    vector<double> grades(n);
    vector<int> credits(n);

    double totalGradePoints = 0;
    int totalCredits = 0;

    // Input course details
    for (int i = 0; i < n; i++) {
        cout << "\nCourse " << i + 1 << ":\n";
        cout << "Enter grade (e.g., 9.0 for A, 8.0 for B, etc.): ";
        cin >> grades[i];
        cout << "Enter credit hours: ";
        cin >> credits[i];

        totalGradePoints += grades[i] * credits[i];
        totalCredits += credits[i];
    }

    // Calculate GPA
    double gpa = 0.0;
    if (totalCredits > 0) {
        gpa = totalGradePoints / totalCredits;
    }

    double cgpa = gpa; // Assuming one semester only (extendable to multiple semesters)

    // Display results
    cout << "\n--- Semester Report ---\n";
    for (int i = 0; i < n; i++) {
        cout << "Course " << i + 1 << ": Grade = " << grades[i]
             << ", Credit Hours = " << credits[i] << "\n";
    }

    cout << "\nTotal Credits: " << totalCredits << endl;
    cout << "Total Grade Points: " << totalGradePoints << endl;
    cout << "GPA this semester: " << gpa << endl;
    cout << "Overall CGPA: " << cgpa << endl;

    return 0;
}
